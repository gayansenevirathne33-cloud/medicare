    <script>
        let inventory = []; let waitingQueue = []; let clinicalArchive = [];
        let activePatient = null; let tempPrescription = []; let editStockId = null; let activeModalRecord = null;

        setInterval(() => { document.getElementById('live-clock').innerText = new Date().toLocaleString(); }, 1000);

        // Core Cloud Listeners
        setTimeout(() => {
            if(!window.db) { alert("Firebase initialization error! Check internet connection."); return; }
            
            // 1. Live Queue Listener
            window.onSnapshot(window.query(window.collection(window.db, "queue"), window.orderBy("token", "asc")), (snapshot) => {
                waitingQueue = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                renderAll();
            });

            // 2. Live Inventory Listener
            window.onSnapshot(window.collection(window.db, "inventory"), (snapshot) => {
                inventory = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                renderAll();
            });

            // 3. Live Archive Listener
            window.onSnapshot(window.query(window.collection(window.db, "archive"), window.orderBy("timestamp", "desc")), (snapshot) => {
                clinicalArchive = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                renderAll();

        function addDrugToPrescription() {
            if(!activePatient) return;
            const select = document.getElementById('doc-drug-select');
            const drugId = select.value; const qtyNeeded = parseInt(document.getElementById('doc-drug-qty').value);
            const dose = document.getElementById('doc-drug-dose').value.trim();
            const freq = document.getElementById('doc-drug-freq').value;
            if(!drugId || !qtyNeeded || !dose) return;
            const stockItem = inventory.find(i => i.id === drugId);
            if(qtyNeeded > stockItem.qty) { alert("Low stock!"); return; }

            tempPrescription.push({
                drugId, name: stockItem.name, dose, frequency: freq, qty: qtyNeeded,
                unitPrice: parseFloat(stockItem.unitPrice), totalCost: qtyNeeded * parseFloat(stockItem.unitPrice)
            });
            document.getElementById('doc-drug-dose').value = ''; document.getElementById('doc-drug-qty').value = '';
            renderTempPrescriptionList(); calculateLiveBillTotals();
        }

        function renderTempPrescriptionList() {
            const list = document.getElementById('doc-selected-drugs');
            if(tempPrescription.length === 0) { list.innerHTML = '<tr><td colspan="4" class="text-center p-2 text-gray-400">No drugs prescribed.</td></tr>'; return; }
            list.innerHTML = tempPrescription.map((d, i) => `<tr><td class="p-2 font-medium">${d.name}</td><td>${d.dose} | ${d.frequency}</td><td class="font-bold">${d.qty}</td><td class="text-center"><button type="button" onclick="removeTempDrug(${i})" class="text-red-500 font-bold">✕</button></td></tr>`).join('');
        }

        window.removeTempDrug = function(i) { tempPrescription.splice(i, 1); renderTempPrescriptionList(); calculateLiveBillTotals(); }

        function calculateLiveBillTotals() {
            const docFee = parseFloat(document.getElementById('bill-doc-fee').value) || 0;
            const hospFee = parseFloat(document.getElementById('bill-hosp-fee').value) || 0;
            const drugsFee = tempPrescription.reduce((sum, item) => sum + item.totalCost, 0);
            const netSubtotal = docFee + hospFee + drugsFee;
            const roundedGrandTotal = Math.ceil(netSubtotal / 50) * 50;
            document.getElementById('bill-rounded-lbl').innerText = roundedGrandTotal.toFixed(2);
            return { docFee, hospFee, drugsFee, netSubtotal, roundedGrandTotal };
        }

        window.saveConsultation = async function(e) {
            e.preventDefault(); if(!activePatient) return;
            const billing = calculateLiveBillTotals();

            // Cloud Inventory Reduction Logic
            for (let p of tempPrescription) {
                const st = inventory.find(i => i.id === p.drugId);
                if(st) {
                    const newQty = st.qty - p.qty;
                    await window.updateDoc(window.doc(window.db, "inventory", p.drugId), { qty: newQty });
                }
            }

            const record = {
                date: new Date().toLocaleDateString('en-US'), time: new Date().toLocaleTimeString(), timestamp: Date.now(),
                token: activePatient.token, name: activePatient.name, age: activePatient.age, gender: activePatient.gender,
                weight: activePatient.weight, contact: activePatient.contact, allergies: activePatient.allergies,
                history: document.getElementById('d-history').value || 'N/A', examination: document.getElementById('d-examination').value || 'N/A',
                investigations: document.getElementById('d-investigations').value || 'N/A', diagnosis: document.getElementById('d-diagnosis').value || 'Unspecified',
                medications: [...tempPrescription], billing
            };

            try {
                await window.addDoc(window.collection(window.db, "archive"), record);
                await window.deleteDoc(window.doc(window.db, "queue", activePatient.id));
                openPrintModal(record); resetActiveConsultation(); document.getElementById('doctor-consult-form').reset();
            } catch(err) { alert("Error checking out: " + err.message); }
        }

        function calculateFinanceMetrics() {
            const todayStr = new Date().toLocaleDateString('en-US'); 
            const currentMonthIdx = new Date().getMonth(); 
            const currentYear = new Date().getFullYear();
            let todaySum = 0; let monthSum = 0;
            const tbody = document.getElementById('finance-table-body'); tbody.innerHTML = clinicalArchive.length === 0 ? '<tr><td colspan="6" class="text-center p-4 text-gray-400">No transactions recorded.</td></tr>' : '';

            clinicalArchive.forEach(r => {
                const total = r.billing.roundedGrandTotal; 
                if(r.date === todayStr) todaySum += total;
                
                const parts = r.date.split('/'); 
                if(parts.length === 3) { 
                    if((parseInt(parts[0]) - 1) === currentMonthIdx && parseInt(parts[2]) === currentYear) monthSum += total; 
                }
                tbody.innerHTML += `<tr class="border-b text-xs"><td class="p-3">${r.date} <span class="text-[10px] text-gray-400">${r.time}</span></td><td class="p-3 font-bold">${r.name}</td><td class="p-3 font-mono">Rs. ${r.billing.docFee.toFixed(2)}</td><td class="p-3 font-mono">Rs. ${r.billing.hospFee.toFixed(2)}</td><td class="p-3 font-mono">Rs. ${r.billing.drugsFee.toFixed(2)}</td><td class="p-3 font-mono font-bold text-right text-indigo-950">Rs. ${total.toFixed(2)}</td></tr>`;
            });
            document.getElementById('fin-today-total').innerText = todaySum.toFixed(2); document.getElementById('fin-month-total').innerText = monthSum.toFixed(2); document.getElementById('fin-visit-count').innerText = clinicalArchive.length;
        }
